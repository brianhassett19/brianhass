Chapter 5 Resampling Methods

This notebook is a rough outline of Chapter 5 of An Introduction to Statistical Learning (James et al. (2013)1).

The authors do not provide code for the analysis or figures used throughout each chapter. They do however provide access to some of the data. This notebook uses the data to recreate some of the analysis and figures from the chapter.

# Left panel:
set.seed(1)
test <- sample(nrow(Auto), nrow(Auto)/2)
degrees <- 1:10

mods <- map(degrees, function(a) {
  mod <- lm(mpg ~ poly(horsepower, a), data = Auto[-test, ])
})

MSE <- map_dbl(mods, function(mod) {
  preds <- predict(mod, newdata = Auto[test, ])
  MSE <-  mean((preds - Auto$mpg[test])^2)
  return(MSE)
})

# Need to make the RHS before we can add the scales

# Right panel:
set.seed(5) # depending on the seed the plots looke very differen, and not very like the text
k = 10 # no. of iterations

MSE_k <- matrix(0, nrow = k, ncol = length(degrees)) # MSE for each iteration

for(i in 1:k) {
  test <- sample(nrow(Auto), nrow(Auto)/2)
  mods <- map(degrees, function(a) {
    mod <- lm(mpg ~ poly(horsepower, a), data = Auto[-test, ])
  })
  
  MSE_k[i, ] <- map_dbl(mods, function(mod) {
    preds <- predict(mod, newdata = Auto[test, ])
    MSE <-  mean((preds - Auto$mpg[test])^2)
    return(MSE)
  })
}

p2 <- data.frame(degree = degrees, MSE_k) %>% 
  pivot_longer(cols = X1:X10, names_to = "iteration", values_to = "MSE") %>% 
  ggplot() +
  geom_line(aes(degree, MSE, color = iteration),
            show.legend = FALSE) +
  scale_x_continuous(name = "Degree of Polynomial",
                     breaks = seq(2, 10, by = 2)) +
  scale_y_continuous(name = "Mean Squared Error",
                     breaks = seq(round(min(MSE_k)), round(max(MSE_k)), by = 2)) +
  ggsci::scale_color_d3() +
  theme(panel.grid = element_blank())

colors_5.2 <- c("red" = "#EB1F2B")

p1 <- tibble(
  degree = degrees,
  MSE = MSE
  ) %>% 
  ggplot(aes(degree, MSE)) +
  geom_point(color = colors_5.2["red"]) +
  geom_line(color = colors_5.2["red"]) +
  scale_x_continuous(name = "Degree of Polynomial",

colors_5.4 <- c("blue" = "#37409E")

# Left panel:
set.seed(1)
d <- 10
cv_error <- rep(0, d)

for(i in 1:d) {
  mod <- glm(mpg ~ poly(horsepower, i), data = Auto)
  cv_mod <- cv.glm(Auto, mod)
  cv_error[i] <- cv_mod$delta[1]
}

# Need to make the RHS before we can add the scales

# Right panel:
set.seed(1)
iterations <- 9
d <- 10
k <- 10

cv_errors <- matrix(0, ncol = d, nrow = iterations)

for(i in 1:iterations) {
  for(j in 1:d) {
    mod <- glm(mpg ~ poly(horsepower, j), data = Auto)
    cv_mod <- cv.glm(Auto, mod, K = k)
    cv_errors[i, j] <- cv_mod$delta[1]
  }
}

p1 <- tibble(
  d = 1:d,
  cv_error
) %>% 
  ggplot(aes(d, cv_error)) +
  geom_point(color = colors_5.4["blue"]) +
  geom_line(color = colors_5.4["blue"]) +
  labs(x = "Degree of Polynomial", y = "Mean Squared Error") +
  scale_x_continuous(breaks = seq(2, 10, by = 2)) +
  scale_y_continuous(breaks = seq(16, 28, by = 2),
                     limits = c(16, 28)) +
  theme(panel.grid = element_blank())

p2 <- data.frame(iteration = 1:iterations, cv_errors) %>% 
  pivot_longer(cols = X1:X10, names_to = "degree", values_to = "MSE") %>% 
  mutate(degree = str_replace(degree, "X", ""),
         degree = as.numeric(degree),
         iteration = factor(iteration)) %>% 
  ggplot(aes(degree, MSE, color = iteration)) +
  geom_line(show.legend = FALSE) +
  labs(x = "Degree of Polynomial", y = "Mean Squared Error") +
  scale_x_continuous(breaks = seq(2, 10, by = 2)) +
  scale_y_continuous(breaks = seq(16, 28, by = 2),
                     limits = c(16, 28)) +

colors_5.9 <- c("green" = "#249968")

bootstrap_sample <- function(data) {
  n <- nrow(data)
  sample <- sample(1:n, n, replace = TRUE)
  return(data[sample, ])
}

G <- vector("list", 4)

colors_5.10 <- c("blue" = "#68A5DB",
                "orange" = "#D99411",
                "grey" = "#231F20")
alpha <- rep(0, 1000)

set.seed(1)
for(i in 1:1000) {
  sample <- bootstrap_sample(Portfolio)
  var.X <- var(sample$X)
  var.Y <- var(sample$Y)
  var.XY <- cov(sample$X, sample$Y)
  alpha[i] <- (var.Y - var.XY)/(var.X + var.Y - 2*var.XY)
}

tibble(alpha) %>%
